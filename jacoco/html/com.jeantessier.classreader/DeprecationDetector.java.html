package com.jeantessier.classreader;

/**
 * Tells whether a programming element has either a
 * {@link Deprecated @Deprecated} annotation or a {@code @deprecated} javadoc
 * tag.
 */
public class DeprecationDetector extends VisitorBase {
    private static final String DEPRECATED_ANNOTATION = Deprecated.class.getName();

    private boolean deprecated;

    public boolean isDeprecated() {
        return deprecated;
    }

    public void visitClassfile(Classfile classfile) {
        visitClassfileAttributes(classfile);
    }

    public void visitAnnotation(Annotation helper) {
        if (DEPRECATED_ANNOTATION.equals(helper.getType())) {
            deprecated = true;
        }
    }

    public void visitDeprecated_attribute(Deprecated_attribute attribute) {
        deprecated = true;
    }
}
