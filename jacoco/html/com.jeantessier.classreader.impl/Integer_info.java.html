package com.jeantessier.classreader.impl;

import java.io.*;

import com.jeantessier.classreader.*;

public class Integer_info extends ConstantPoolEntry implements com.jeantessier.classreader.Integer_info {
    private final int value;

    public Integer_info(ConstantPool constantPool, DataInput in) throws IOException {
        super(constantPool);

        value = in.readInt();
    }

    public int getValue() {
        return value;
    }

    public String toString() {
        return String.valueOf(getValue());
    }

    public int hashCode() {
        return Integer.valueOf(getValue()).hashCode();
    }

    public boolean equals(Object object) {
        if (this == object) {
            return true;
        }

        if (object == null || getClass() != object.getClass()) {
            return false;
        }

        return getValue() == ((Integer_info) object).getValue();
    }

    public void accept(Visitor visitor) {
        visitor.visitInteger_info(this);
    }
}
